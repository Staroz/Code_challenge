//date: 08/27/2023
 
function solution(number){
  let count = 0;
  if (number < 3) return count;
  for (let i = 3; i < number; i++) {
    if (i%3 ===0) {
      count+=i;
    } else if (i%5 ===0) {
      count+=i;
    }
  };
  return count
}
 
 
module.exports = {
  solution,
}
